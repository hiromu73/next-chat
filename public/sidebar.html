以下に、コードをMUIを使用してUIを調整し、コンポーネントを分割した構成を示します。

構成概要

  1. components/ChatUI.tsx: 全体のレイアウト管理。
  2. components/Sidebar.tsx: サイドバー部分を独立。
  3. components/Header.tsx: ヘッダー部分を独立。
  4. components/ChatArea.tsx: チャット表示領域。
  5. components/ChatInput.tsx: 入力フォーム。

ChatUI.tsx

tsx
'use client';

import { useState } from 'react';
import Sidebar from './Sidebar';
import Header from './Header';
import ChatArea from './ChatArea';
import ChatInput from './ChatInput';
import { Box, CssBaseline } from '@mui/material';

export default function ChatUI() {
  const [sidebarOpen, setSidebarOpen] = useState(false);
  const [messages, setMessages] = useState([
    { role: 'assistant', content: 'こんにちは！どのようなお手伝いができますか？' },
  ]);

  const toggleSidebar = () => setSidebarOpen(!sidebarOpen);

  const handleSendMessage = (input: string) => {
    setMessages((prev) => [
      ...prev,
      { role: 'user', content: input },
      { role: 'assistant', content: '申し訳ありませんが、これはデモUIです。実際の応答は生成されません。' },
    ]);
  };

  return (
    <Box sx={{ display: 'flex', height: '100vh', bgcolor: 'grey.100' }}>
      <CssBaseline />
      <Sidebar open={sidebarOpen} onToggle={toggleSidebar} />
      <Box sx={{ flex: 1, display: 'flex', flexDirection: 'column' }}>
        <Header onToggleSidebar={toggleSidebar} />
        <ChatArea messages={messages} />
        <ChatInput onSendMessage={handleSendMessage} />
      </Box>
    </Box>
  );
}

ChatArea.tsx

tsx
import { Box, Card, Typography } from '@mui/material';

interface Message {
  role: 'user' | 'assistant';
  content: string;
}

interface ChatAreaProps {
  messages: Message[];
}

export default function ChatArea({ messages }: ChatAreaProps) {
  return (
    <Box sx={{ flex: 1, overflowY: 'auto', p: 2 }}>
      <Card sx={{ maxWidth: '600px', mx: 'auto', p: 2 }}>
        {messages.map((message, index) => (
          <Box
            key={index}
            sx={{
              display: 'flex',
              justifyContent: message.role === 'user' ? 'flex-end' : 'flex-start',
              mb: 1,
            }}
          >
            <Typography
              variant="body1"
              sx={{
                bgcolor: message.role === 'user' ? 'primary.main' : 'grey.200',
                color: message.role === 'user' ? 'white' : 'black',
                borderRadius: 1,
                p: 1,
                maxWidth: '70%',
              }}
            >
              {message.content}
            </Typography>
          </Box>
        ))}
      </Card>
    </Box>
  );
}

ChatInput.tsx

tsx
import { Box, TextField, Button } from '@mui/material';
import { useState } from 'react';

interface ChatInputProps {
  onSendMessage: (input: string) => void;
}

export default function ChatInput({ onSendMessage }: ChatInputProps) {
  const [input, setInput] = useState('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (input.trim()) {
      onSendMessage(input);
      setInput('');
    }
  };

  return (
    <Box
      component="form"
      onSubmit={handleSubmit}
      sx={{ p: 2, bgcolor: 'white', borderTop: '1px solid grey.300' }}
    >
      <Box sx={{ display: 'flex', maxWidth: 600, mx: 'auto' }}>
        <TextField
          value={input}
          onChange={(e) => setInput(e.target.value)}
          fullWidth
          placeholder="メッセージを入力..."
        />
        <Button type="submit" variant="contained" sx={{ ml: 1 }}>
          送信
        </Button>
      </Box>
    </Box>
  );
}

この分割により、UIが整理され、コンポーネントの再利用やメンテナンスが容易になります。また、MUIを使用して柔軟なスタイリングを提供します。
